------------------------------------
【push時にユーザー名とパスワードを省略する方法】
{cloneしたリポジトリDir}/.git/configを編集する
[remote "origin"]
	url = https://username:password@github.com/username/repository.git
------------------------------------
【基本コマンド】
git clean [repository]
git add [file name]
git commt -m "comment"
git push origin main
